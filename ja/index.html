phiori

伺かのためのpython3基盤のSHIORI

たやすいイベントハンドリング

@handle("OnBoot")
def boot():
	write(r"\1\s0\0\s0元気だった？\e")

簡単なランダムセリフ

@handle("OnGhostChanged")
def ghostchanged(*args):
	write(r"\1\s0\0\s0")
	write(
		[
			"{}さん、お疲れさま！",
			"{}さんと交代しました！"
		],
		*args
	)
	write(r"\e")

便利な固有イベント

@handle("OnStroke")
def stroke(*args):
	if args[1] == "Head":
		write(r"\0\s1気持ち……\w6いいね……。\e")
