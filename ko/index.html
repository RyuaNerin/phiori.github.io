phiori

우카가카를 위한 python3 기반의 SHIORI

손쉬운 이벤트 핸들링

@handle("OnBoot")
def boot():
	write(r"\1\s0\0\s0잘 지냈어?\e")

간편한 임의 대사

@handle("OnGhostChanged")
def ghostchanged(*args):
	write(r"\1\s0\0\s0")
	write(
		[
			"{}씨, 수고 많았어!",
			"{}씨랑 교대했어!"
		],
		*args
	)
	write(r"\e")

편리한 고유 이벤트

@handle("OnStroke")
def stroke(*args):
	if args[1] == "Head":
		write(r"\0\s1기분은…… \w6좋네…….\e")
